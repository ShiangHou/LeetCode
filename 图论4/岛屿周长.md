from collections import deque

def bfs(graph,i,j,visited,direction,n,m):
    dq = deque()
    #先把第一个加进去
    dq.append(graph[i][j])
    #注意，如果陆地共用0，只用计算一次，所以0的visited也要标记一下
    #这里的流程是，弹出一个，计算这一个的四周有多少0，然后统计进去周长并标记visted，有多少1，加入队列
    temp_length = 0
    while dq:
        #先弹
        temp_node = dq.popleft()

        #弹出的遍历四个方向
        for x,y in direction:
            next_i,next_j = x+i,y+j
            #如果出界了，不用担心重合，只有像两个犄角一样的才会重合
            if next_i <0 or next_i>n-1 or next_j<0 or next_j>m-1:
                temp_length += 1#外面也是水，也算周长
            else:#没出界
                #如果是陆地，直接加入队列就行
                if graph[next_i][next_j] == 1 and visited[next_i][next_j] == False:
                    dq.append(graph[next_i][next_j])
                    #加进去就算记录了，记录visted
                    visited[next_i][next_j] = True
                elif graph[next_i][next_j] == 0:# and visited[next_i][next_j] == False:#没遇到的0
                    temp_length += 1#统计周长
                    #visited[next_i][next_j] = True#记录这个0统计过了
    return temp_length
            

def main():
    n,m  = map(int,input().split())
    graph = []

    for i in range(n):
        graph.append(list(list(map(int,input().split()))))


    direction = [[1,0],[-1,0],[0,1],[0,-1]]

    visited = [[False]*m for _ in range(n)]

    length = 0

    for i in range(n):

            if graph[i][j] == 1 and visited[i][j] == False:
                length += bfs(graph,i,j,visited,direction,n,m)#广搜返回的是面积
    
    return length

if __name__ == "__main__":
