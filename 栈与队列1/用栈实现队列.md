## 初步思路
这题之前是做过，但是有点忘记了怎么在python中去生成一个栈

在python中，我们可以用一个list来去模拟一个栈，append是压入，pop是弹出，[-1]是查看栈顶

用两个栈去模拟队列，我依旧有印象

首先是入，正常压入一个栈就好

其次是出，出的时候有讲究，先把入队列统统倒入到出栈，然后弹出，

```python 
class MyQueue:
    def __init__(self):
        self.in_stack = []
        self.out_stack = []
        self.size = 0
        

    def push(self, x: int) -> None:
        self.in_stack.append(x)
        self.size += 1

    def pop(self) -> int:
        if self.out_stack:#如果out不是空的，那么直接从out弹出一个就行s
            self.size -= 1
            return self.out_stack.pop()
        else:#如果是空，就把in全部倒出来到out
            while self.in_stack:
                self.out_stack.append(self.in_stack.pop())
            self.size -= 1
            return self.out_stack.pop()

    def peek(self) -> int:
        if self.out_stack:
            return self.out_stack[-1]
        elif self.in_stack:
            while self.in_stack:
                self.out_stack.append(self.in_stack.pop())
            return self.out_stack[-1]
        else:
            return False

    def empty(self) -> bool:
        return self.size == 0

```
